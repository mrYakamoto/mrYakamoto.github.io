Ruby Enumberables Cheat Sheet

Most used

#each

each { |item| block } --> ary

#map

map { |obj| block } --> array

#all?

all? [{ |obj| block } ] --> true or false

#any?

any? [{ |obj| block }] --> true or false

#detect / #find

detect(ifnone = nil) { |obj| block } --> obj or nil

#each_slice

(1..6).each_slice(3) { |a| p a } -->
[1, 2, 3]
[4, 5, 6]

#each_with_index

each_with_index(*args) { |obj, i| block } --> enum click to toggle source

#grep

grep(pattern) --> array
grep(pattern) { |obj| block } --> array

#max

max --> obj click to toggle source
max { |a, b| block } --> obj
max(n) --> obj
max(n) {|a,b| block } --> obj

#min

min --> obj
min {| a,b | block } --> obj
min(n) --> array
min(n) {| a,b | block } --> array

#reduce

reduce(initial, sym) --> obj
reduce(sym) --> obj
reduce(initial) { |memo, obj| block } --> obj
reduce { |memo, obj| block } --> obj

#sort

sort --> array
sort { |a, b| block } --> array

Less used

#chunk

chunk { | elt | ... } --> an_enumerator
chunk(initial_state) { |elt, state| ... } --> an_enumerator (deprecated)

#collect

collect_concat { |obj| block } --> array
collect_concat --> an_enumerator

#count

count --> int
count(item) --> int
count { |obj| block } --> int

#cycle

cycle(n=nil) { |obj| block } --> nil
cycle(n=nil) --> an_enumerator

#drop

drop(n) --> array

#each_cons

each_cons(n) { ... } --> nil
each_cons(n) → an_enumerator

#each_entry

each_entry { |obj| block } --> enum
each_entry --> an_enumerator

#each_with_object

each_with_object(obj) { |(*args), memo_obj| ... } --> obj
each_with_object(obj) --> an_enumerator

#entries

entries(*args) --> array

#find_all

find_all { |obj| block } --> array
find_all --> an_enumerator

#find_index

find_index(value) --> int or nil
find_index { |obj| block } --> int or nil
find_index --> an_enumerator

#first

first --> obj or nil
first(n) --> an_array

#flat_map

flat_map { |obj| block } --> array
flat_map --> an_enumerator

#include?

include?(obj) --> true or false

#inject

inject(initial, sym) --> obj
inject(sym) --> obj
inject(initial) { |memo, obj| block } --> obj
inject { |memo, obj| block } --> obj

#lazy

lazy --> lazy_enumerator

#min_by

min_by {|obj| block } --> obj
min_by --> an_enumerator
min_by(n) {|obj| block } --> array
min_by(n) --> an_enumerator

#minmax

minmax --> [min, max]
minmax { |a, b| block } --> [min, max]

#none?

none? [{ |obj| block }] --> true or false

#one?

one? [{ |obj| block }] --> true or false

#partition

partition { |obj| block } --> [ true_array, false_array ]
partition → an_enumerator

#reject

reject { |obj| block } --> array
reject → an_enumerator

#reverse_each

reverse_each(*args) { |item| block } --> enum
reverse_each(*args) --> an_enumerator

#slice_after

slice_after(pattern) --> an_enumerator
slice_after { |elt| bool } --> an_enumerator

#slice_before

slice_before(pattern) --> an_enumerator
slice_before { |elt| bool } --> an_enumerator
slice_before(initial_state) { |elt, state| bool } --> an_enumerator (deprecated)

#slice_when

slice_when {|elt_before, elt_after| bool } --> an_enumerator

#to_a

to_a(*args) --> array

#to_h

to_h(*args) --> hash

#zip

zip(arg, ...) --> an_array_of_array
zip(arg, ...) { |arr| block } --> nil

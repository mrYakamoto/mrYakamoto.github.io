Ruby Enumerables

#each -

calls the given block once for each element in self, passing that element as a parameter.
each { |item| block }

#all?

returns true if any element passes the given code block
all? { |object| block }

any?

returns true if any element passes the given code block
any? { |object| block }

#collect

returns a new array with the result of passing a given code block on each element
collect { |object| block }

#each_slice

iterates the given block for each slice of n elements.
each_slice(n) { |n objects| block }

#each_with_index

calls given block with two arguments, the item and its index.
each_with_index { |obj, i| block }

#find / #detect

passes each entry to the given block. Returns the first for which block returns true.
find { |obj| block }

#find_all

passes each entry to given block. Returns an array containing all that return true.
find_all { |obj| block }

#find_index

passes each entry to given block and returns the index for the first which returns true.
#find_index { |obj| block }

#grep

returns an array of every element which pattern === element. If optional block is passed to it, each matching element is put through block and the block's result is stored in returned array.
grep(/regex/) { |obj| optional block }
